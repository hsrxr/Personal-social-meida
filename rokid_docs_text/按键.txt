# Rokid Glasse 上的系统功能按键

在Rokid Glasses 上，可以通过监听有序广播来截止按键的系统事件触发。

## 1 按键广播

具体的按键广播信息参考如下：

package com.rokid.cxrssdksamples.activities.keys

import android.content.BroadcastReceiver
import android.content.Context
import android.content.Intent

interface KeyReceiverListener {
    fun onReceive(keyType: KeyType)
}

enum class KeyType(val action: String) {
    CLICK(&quot;com.android.action.ACTION_SPRITE_BUTTON_CLICK&quot;),
    BUTTON_DOWN(&quot;com.android.action.ACTION_SPRITE_BUTTON_DOWN&quot;),
    BUTTON_UP(&quot;com.android.action.ACTION_SPRITE_BUTTON_UP&quot;),
    DOUBLE_CLICK(&quot;com.android.action.ACTION_SPRITE_BUTTON_DOUBLE_CLICK&quot;),
    AI_START(&quot;com.android.action.ACTION_AI_START&quot;),
    LONG_PRESS(&quot;com.android.action.ACTION_SPRITE_BUTTON_LONG_PRESS&quot;),
    ACTION_TWO_FINGER_SINGLE_TAP(&quot;com.android.action.ACTION_TWO_FINGER_SINGLE_TAP&quot;),
    ACTION_TWO_FINGER_DOUBLE_TAP(&quot;com.android.action.ACTION_TWO_FINGER_DOUBLE_TAP&quot;),
    ACTION_TWO_FINGER_SWIPE_FORWARD(&quot;com.android.action.ACTION_TWO_FINGER_SWIPE_FORWARD&quot;),
    ACTION_TWO_FINGER_SWIPE_BACK(&quot;com.android.action.ACTION_TWO_FINGER_SWIPE_BACK&quot;),
    ACTION_SETTINGS_KEY(&quot;com.android.action.ACTION_SETTINGS_KEY&quot;)
}

class KeyReceiver : BroadcastReceiver() {

    var listener: KeyReceiverListener? = null

    override fun onReceive(context: Context?, intent: Intent?) {
        intent?.action?.let {
            when (it) {
                KeyType.CLICK.action -&gt; {
                    // 接收到了按键点击--截止广播
                    listener?.onReceive(KeyType.CLICK)
                    abortBroadcast()
                }
                KeyType.BUTTON_DOWN.action -&gt; {
                    // 监听按键按下--截止广播
                    listener?.onReceive(KeyType.BUTTON_DOWN)
                    abortBroadcast()
                }
                KeyType.BUTTON_UP.action -&gt; {
                    // 监听按键抬起--截止广播
                    listener?.onReceive(KeyType.BUTTON_UP)
                    abortBroadcast()
                }
                KeyType.DOUBLE_CLICK.action -&gt; {
                    // 监听按键双击--截止广播--不可截断--该事件为退出
                    listener?.onReceive(KeyType.DOUBLE_CLICK)
                    abortBroadcast()
                }
                KeyType.AI_START.action -&gt; {
                    // 监听AI开始--截止广播
                    listener?.onReceive(KeyType.AI_START)
                    abortBroadcast()
                }
                KeyType.LONG_PRESS.action -&gt; {
                    // 监听按键长按--截止广播
                    listener?.onReceive(KeyType.LONG_PRESS)
                    abortBroadcast()
                }
                KeyType.ACTION_TWO_FINGER_SINGLE_TAP.action -&gt; {
                    // 监听按键双指单机--截止广播
                    listener?.onReceive(KeyType.ACTION_TWO_FINGER_SINGLE_TAP)
                    abortBroadcast()
                }
                KeyType.ACTION_TWO_FINGER_DOUBLE_TAP.action -&gt; {
                    // 监听按键双击--截止广播
                    listener?.onReceive(KeyType.ACTION_TWO_FINGER_DOUBLE_TAP)
                    abortBroadcast()
                }
                KeyType.ACTION_TWO_FINGER_SWIPE_FORWARD.action -&gt; {
                    // 监听按键双击--截止广播
                    listener?.onReceive(KeyType.ACTION_TWO_FINGER_SWIPE_FORWARD)
                    abortBroadcast()
                }
                KeyType.ACTION_TWO_FINGER_SWIPE_BACK.action -&gt; {
                    // 监听按键双击--截止广播
                    listener?.onReceive(KeyType.ACTION_TWO_FINGER_SWIPE_BACK)
                    abortBroadcast()
                }
                KeyType.ACTION_SETTINGS_KEY.action -&gt; {
                    // 监听按键双击--截止广播
                    listener?.onReceive(KeyType.ACTION_SETTINGS_KEY)
                    abortBroadcast()
                }
            }
        }
    }
}

2 按键广播的使用
参考如下代码：

private val keyReceiver = KeyReceiver().apply {
    listener = object : KeyReceiverListener {
        override fun onReceive(keyType: KeyType) {
            // 更新状态以在UI中显示最新的按键类型
            latestKeyType = keyType

            when (keyType) {
                KeyType.CLICK -&gt; {
                    // 处理按键点击事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: button on left leg&quot;)
                }
                KeyType.BUTTON_DOWN -&gt; {
                    // 处理按键按下事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: button on left leg down&quot;)
                }
                KeyType.BUTTON_UP -&gt; {
                    // 处理按键抬起事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: button on left leg up&quot;)
                }
                KeyType.DOUBLE_CLICK -&gt; {
                    // 处理按键双击事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: button on left leg double click&quot;)
                }
                KeyType.AI_START -&gt; {
                    // 处理touchpad 长按事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: touchpad long pressed&quot;)
                }
                KeyType.LONG_PRESS -&gt; {
                    // 处理按键长按事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: long pressed the button on left leg&quot;)
                }

                KeyType.ACTION_TWO_FINGER_SINGLE_TAP -&gt; {
                    // 处理双指单击事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: two finger single tap&quot;)
                }
                KeyType.ACTION_TWO_FINGER_DOUBLE_TAP -&gt; {
                    // 处理双指双击事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: two finger double tap&quot;)
                }
                KeyType.ACTION_TWO_FINGER_SWIPE_FORWARD -&gt; {
                    // 处理双指滑动事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: two finger swipe forward&quot;)
                }
                KeyType.ACTION_TWO_FINGER_SWIPE_BACK -&gt; {
                    // 处理双指滑动事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: two finger swipe back&quot;)
                }
                KeyType.ACTION_SETTINGS_KEY -&gt; {
                    // 处理双指长按事件
                    Log.d(&quot;KeysActivity&quot;, &quot;system event: two finger long pressed&quot;)
                }
            }
        }
    }
}
@SuppressLint(&quot;UnspecifiedRegisterReceiverFlag&quot;)
override fun onCreate(savedInstanceState: Bundle?) {
    super.onCreate(savedInstanceState)
    enableEdgeToEdge()

    // 设置屏幕常亮
    window.addFlags(android.view.WindowManager.LayoutParams.FLAG_KEEP_SCREEN_ON)

    setContent {
        CXRSSDKSamplesTheme {
            KeysScreen(latestKeyType = latestKeyType?.name ?: &quot;&quot;)
        }
    }
    registerReceiver(keyReceiver, IntentFilter().apply {
        addAction(KeyType.CLICK.action)
        addAction(KeyType.BUTTON_DOWN.action)
        addAction(KeyType.BUTTON_UP.action)
        addAction(KeyType.DOUBLE_CLICK.action)
        addAction(KeyType.AI_START.action)
        addAction(KeyType.LONG_PRESS.action)
        addAction(KeyType.ACTION_TWO_FINGER_SINGLE_TAP.action)
        addAction(KeyType.ACTION_TWO_FINGER_DOUBLE_TAP.action)
        addAction(KeyType.ACTION_TWO_FINGER_SWIPE_FORWARD.action)
        addAction(KeyType.ACTION_TWO_FINGER_SWIPE_BACK.action)
        addAction(KeyType.ACTION_SETTINGS_KEY.action)
        priority = 100
    })
}

其他按键
其他的按键信息都可以通过监听系统keyEvent 来监听。

@SuppressLint(&quot;GestureBackNavigation&quot;)
override fun onKeyDown(keyCode: Int, event: KeyEvent?): Boolean {
    Log.d(&quot;KeysActivity&quot;, &quot;onKeyDown: $keyCode&quot;)
    when (keyCode) {
        KeyEvent.KEYCODE_BACK -&gt; {
            // 拦截返回键
            Log.d(&quot;KeysActivity&quot;, &quot;onKeyDown: back pressed&quot;)
            return true
        }
        KeyEvent.KEYCODE_ENTER-&gt;{
            Log.d(&quot;KeysActivity&quot;, &quot;onKeyUp: touchpad single down&quot;)
            return  true
        }
        else -&gt; {
            Log.d(&quot;KeysActivity&quot;, &quot;onKeyUp: $keyCode&quot;)
        }
    }
    return super.onKeyDown(keyCode, event)
}

@SuppressLint(&quot;GestureBackNavigation&quot;)
override fun onKeyUp(keyCode: Int, event: KeyEvent?): Boolean {
    Log.d(&quot;KeysActivity&quot;, &quot;onKeyUp: $keyCode&quot;)
    when (keyCode) {
        KeyEvent.KEYCODE_BACK -&gt; {
            // 拦截返回键
            Log.d(&quot;KeysActivity&quot;, &quot;onKeyUp: back pressed&quot;)
            return true
        }
        KeyEvent.KEYCODE_ENTER-&gt;{
            Log.d(&quot;KeysActivity&quot;, &quot;onKeyUp: touchpad single up&quot;)
            return  true
        }
        else -&gt; {
            Log.d(&quot;KeysActivity&quot;, &quot;onKeyUp: $keyCode&quot;)
        }
    }
    return super.onKeyUp(keyCode, event)
}
